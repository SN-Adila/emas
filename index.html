Senarai Laporan Aktiviti

Panduan Integrasi Google Sheets & Drive

Sambungkan Dashboard ini terus ke Google Sheets & Drive yang telah anda sediakan di Bagan Serai.

Mengapa Integrasi Diperlukan?

Penyemak imbas web (browser) tidak dibenarkan menyimpan data terus ke Google Sheets / Drive atas faktor keselamatan secara langsung. Kita perlu menggunakan Google Apps Script sebagai jambatan (API Webhook) yang menyokong penghantaran data tanpa pelayan tambahan.

Data anda akan dihantar ke URL ini setiap kali anda klik "Simpan Laporan". Jika dibiarkan kosong, ia menggunakan storan tempatan (Local Storage).

Langkah 2: Salin Kod Google Apps Script di bawah:

/* KOD GOOGLE APPS SCRIPT UNTUK INTEGRASI
Sila paste kod ini di Extension > Apps Script di Google Sheet anda.
*/
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.openById("1ReX5B40-6twj1K2mWnUT7wNpKEuVFQC-EHqmr-6wkOw").getActiveSheet();
    
    // Folder Laporan PDF: 1CleTztYyzLrsOvhU36oeEMP2ybJpyaAf
    var folderPdfId = "1CleTztYyzLrsOvhU36oeEMP2ybJpyaAf";
    // Folder Gambar: 1YsM-gEFzGlcZzRPTcjIoLezpn5Kvd9rD
    var folderGambarId = "1YsM-gEFzGlcZzRPTcjIoLezpn5Kvd9rD";
    
    // Simpan PDF Laporan ke Google Drive Folder
    var pdfUrl = "";
    if (data.PDF_BASE64) {
      var cleanPdfData = data.PDF_BASE64.split(",")[1];
      var pdfBlob = Utilities.newBlob(Utilities.base64Decode(cleanPdfData), "application/pdf", "Laporan_" + data.NAMA_AKTIVITI + "_" + data.TARIKH + ".pdf");
      var pdfFile = DriveApp.getFolderById(folderPdfId).createFile(pdfBlob);
      pdfUrl = pdfFile.getUrl();
    }
    
    var imgUrls = ["", "", "", ""];
    for (var i = 1; i <= 4; i++) {
      var base64Key = "GAMBAR_" + i;
      if (data[base64Key] && data[base64Key].indexOf("data:image") > -1) {
        var cleanData = data[base64Key].split(",")[1];
        var contentType = data[base64Key].split(";")[0].split(":")[1];
        var blob = Utilities.newBlob(Utilities.base64Decode(cleanData), contentType, data.NAMA_AKTIVITI + "_Img" + i + ".png");
        var file = DriveApp.getFolderById(folderGambarId).createFile(blob);
        imgUrls[i-1] = file.getUrl();
      }
    }

    sheet.appendRow([
      data.NAMA_AKTIVITI,
      data.TARIKH,
      data.MASA,
      data.TEMPAT,
      data.SASARAN,
      data.BIL_PESERTA,
      data.ANGGOTA_TERLIBAT,
      data.OBJEKTIF,
      data.PELAKSANAAN,
      data.PENCAPAIAN,
      data.ISU,
      data.PENAMBAHBAIKAN,
      imgUrls[0],
      imgUrls[1],
      imgUrls[2],
      imgUrls[3]
    ]);

    return ContentService.createTextOutput(JSON.stringify({status: "success", pdfUrl: pdfUrl}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
